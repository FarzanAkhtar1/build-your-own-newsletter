const SiteRibbon = {
   maxQueueSize: 5, // max amount of times user should see banner within allotted time
   maxTimeInSeconds: 86400, // One day
   showRibbon: false,
   currentTimeInSeconds: Math.round(Date.now() / 1000),
   init: function() {
      // Local Storage names
      const localStorageId = 'nn_site_ribbon_dismiss-birthday';
      const localStorageQueue = 'nn_site_ribbon_birthday-queue';

      // Display values used
      const DISPLAY_TYPES = { NONE: 'none', BLOCK: 'block' };
      let ribbonDisplayValue = DISPLAY_TYPES.NONE;

      // Query selectors 
      const dismissRibbon = document.querySelector('.dismiss-ribbon');
      const siteRibbon = document.querySelector('.site-ribbon');
      const body = document.querySelector('body');             

      // If user has has dismissed the banner then there is no need to do anything
      const userDismissed = window.localStorage.getItem(localStorageId);
      if(userDismissed){
         siteRibbon.style = 'display:' + ribbonDisplayValue + ';';
         return;
      }

      // Check banner queue to see if we should show banner, if so then add to banner queue
      const seenRibbonQueue = JSON.parse(window.localStorage.getItem(localStorageQueue)) || [];
      const filteredRibbonQueue = this.updateBannerQueue(seenRibbonQueue);
      if(filteredRibbonQueue.length < this.maxQueueSize) {
         window.localStorage.setItem(localStorageQueue, JSON.stringify([...filteredRibbonQueue, this.currentTimeInSeconds]));
         this.showRibbon = true;
      }

      if (this.showRibbon) {
         const hasRibbonClassName = 'has-site-ribbon';
         ribbonDisplayValue = DISPLAY_TYPES.BLOCK; 
        
         body.classList.add(hasRibbonClassName);

         // Close button listener
         dismissRibbon.addEventListener('click', function(){
            window.localStorage.setItem(localStorageId, 1);
            window.localStorage.removeItem(localStorageQueue);

            nn.gtm.triggerGenericEvent('Dismissible Banner', '25th anniversary', 'Close');
            
            siteRibbon.style = 'display: none;';
            body.classList.remove(hasRibbonClassName);
         })
      }

      siteRibbon.style = 'display:' + ribbonDisplayValue + ';';
   },
   updateBannerQueue(seenRibbonQueue) {
      return seenRibbonQueue.filter((visitedTimeInSeconds) => {
         if(this.currentTimeInSeconds - visitedTimeInSeconds < this.maxTimeInSeconds ) {
            return visitedTimeInSeconds;
         } 
      });
   }
}
SiteRibbon.init();
